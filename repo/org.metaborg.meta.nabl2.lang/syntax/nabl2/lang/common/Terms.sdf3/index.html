  NaBL2-Wld.Wld   = "_"

  .As       = <<NaBL2-Var>@<NaBL2-Pattern>>
            = NaBL2-OpTerm
            = NaBL2-IntTerm
            = NaBL2-StringTerm

  NaBL2-Patterns         = <<{NaBL2-Pattern ", "}*>>

  .Int = NaBL2-INT

  NaBL2-OpTerm-CF.Op      = NaBL2-OpId-LEX "(" LAYOUT?-CF NaBL2-Patterns-CF LAYOUT?-CF  ")"

  NaBL2-StringChars   = NaBL2-StringChar*
